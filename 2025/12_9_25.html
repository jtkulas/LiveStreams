viewof freq = Inputs.range([0.5, 15], {value: 5, step: 0.05, label: "Frequency"})

{
  const width = 900;
  const height = 460;

  const canvas = html`<canvas width=${width} height=${height}
    style="width:100%; max-width:960px; height:auto; background:#0f172a; border-radius:12px; box-shadow:0 20px 40px rgba(0,0,0,0.4); display:block; margin:2rem auto;"></canvas>`;
  const ctx = canvas.getContext("2d");

  let t = 0;
  function animate() {
    ctx.fillStyle = "rgba(15, 23, 42, 0.15)";
    ctx.fillRect(0, 0, width, height);

    // Main wave
    ctx.strokeStyle = "#c084fc";
    ctx.lineWidth = 5;
    ctx.beginPath();
    for (let i = 0; i <= 600; i++) {
      const x = (i / 600) * width;
      const y = height / 2
        + Math.sin(x / 100 * freq + t) * 90
        + Math.sin(x / 60 * freq * 1.7 - t * 0.8) * 50;
      i === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y);
    }
    ctx.stroke();

    // Glowing tracer dots
    for (let i = 0; i < 15; i++) {
      const x = width * (i / 14);
      const y = height / 2
        + Math.sin(x / 100 * freq + t) * 90
        + Math.sin(x / 60 * freq * 1.7 - t * 0.8) * 50;

      ctx.fillStyle = "#fde047";
      ctx.beginPath();
      ctx.arc(x, y, 10, 0, Math.PI * 2);
      ctx.fill();
    }

    t += 0.04;
    requestAnimationFrame(animate);
  }
  animate();

  // Title that updates live
  html`<div style="text-align:center; margin:1.5rem 0 0.5rem; font-size:2.5rem; color:#e9d5ff; font-weight:600;">
    Frequency = ${freq.toFixed(2)}
  </div>`;

  return canvas;
}
