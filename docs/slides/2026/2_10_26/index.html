heartdf = tibble(
    t = seq(0, 2*pi, pi/60),
    x = 16*sin(t)^3,
    y = 13*cos(t) - 5*cos(2*t) - 2*cos(3*t) - cos(4*t)
)

ggplot(data = heartdf, aes(x, y)) + 
    geom_polygon(aes(group = 1), fill = "red") +
    geom_text(aes(x = 0, y = 0, label = "\u2661\u2764\u2665 Va va voom!!! \u2665\u2764\u2661", parse=TRUE),
