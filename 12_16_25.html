(function () {
  // Create a unique id so multiple blocks can coexist
  const uid = 'ojs-counter-' + Math.random().toString(36).slice(2, 9);

  // The host is the element that contains this script; insert the widget after it
  const host = document.currentScript && document.currentScript.parentNode ? document.currentScript.parentNode : document.body;

  // Build the widget HTML
  const container = document.createElement('div');
  container.id = uid;
  container.style.maxWidth = '720px';
  container.style.margin = '24px auto';
  container.innerHTML = `
    <div style="text-align:center; padding:28px; font-family:system-ui, -apple-system, 'Segoe UI', Roboto, 'Helvetica Neue', Arial;">
      <button id="${uid}-btn" aria-live="polite" style="font-size:18px; padding:10px 20px; cursor:pointer; border-radius:6px; border:1px solid #ccc; background:#fff;">
        Click to +1!
      </button>

      <div id="${uid}-display" style="margin-top:22px;">
        <h2 style="margin:0; font-weight:600;">Count: <strong id="${uid}-value" style="color:#e15759;">0</strong></h2>
        <p id="${uid}-sub" style="color:#666; margin-top:10px;">You've clicked <strong id="${uid}-plural">0</strong> times!</p>
        <small style="color:#999;">(Click the button above to update)</small>
      </div>
    </div>
  `;

  host.appendChild(container);

  // Mutable state
  let count = 0;

  // Elements
  const btn = document.getElementById(`${uid}-btn`);
  const valueEl = document.getElementById(`${uid}-value`);
  const pluralEl = document.getElementById(`${uid}-plural`);
  const subEl = document.getElementById(`${uid}-sub`);

  // Update UI
  function updateUI() {
    valueEl.textContent = String(count);
    pluralEl.textContent = String(count);
    subEl.innerHTML = `You've clicked <strong>${count}</strong> time${count === 1 ? '' : 's'}!`;
    btn.setAttribute('aria-label', `Click to increment. Current count ${count}`);
  }

  // Click handler
  btn.addEventListener('click', () => {
    count += 1;
    updateUI();
  });

  // Ensure keyboard activation is handled (button already does this by default)
  btn.addEventListener('keydown', (e) => {
    if (e.key === 'Enter' || e.key === ' ') {
      e.preventDefault();
      count += 1;
      updateUI();
    }
  });

  // Initial render
  updateUI();
})();

(function () {
  // Unique id for multiple independent blocks
  const uid = 'ojs-scatter-' + Math.random().toString(36).slice(2, 9);

  // Host (insert after the script tag that runs in the ojs block)
  const host = document.currentScript && document.currentScript.parentNode ? document.currentScript.parentNode : document.body;

  // Build UI skeleton
  const wrapper = document.createElement('div');
  wrapper.id = uid;
  wrapper.style.maxWidth = '960px';
  wrapper.style.margin = '18px auto';
  wrapper.style.fontFamily = "system-ui, -apple-system, 'Segoe UI', Roboto, 'Helvetica Neue', Arial";
  wrapper.innerHTML = `
    <div style="display:flex; gap:18px; align-items:flex-start; flex-wrap:wrap;">
      <div style="flex:1 1 360px; min-width:300px;">
        <div style="display:flex; justify-content:space-between; align-items:center; margin-bottom:6px;">
          <strong>Interactive scatter</strong>
          <small style="color:#666;">Drag sliders or hover points</small>
        </div>
        <div id="${uid}-controls" style="background:#fafafa; border:1px solid #eee; padding:12px; border-radius:8px;">
          <label style="display:block; margin-bottom:8px;">
            Points: <span id="${uid}-n-label">200</span>
            <input id="${uid}-n" type="range" min="20" max="2000" step="10" value="200" style="width:100%;">
          </label>

          <label style="display:block; margin-bottom:8px;">
            Min Y filter: <span id="${uid}-ymin-label">-3.0</span>
            <input id="${uid}-ymin" type="range" min="-5" max="5" step="0.1" value="-3" style="width:100%;">
          </label>

          <label style="display:block; margin-bottom:8px;">
            Color mode:
            <select id="${uid}-color" style="margin-left:8px;">
              <option value="single">Single color</option>
              <option value="byCategory">Categorical</option>
              <option value="byX">Color by x (gradient)</option>
            </select>
          </label>

          <label style="display:flex; gap:8px; align-items:center;">
            <input id="${uid}-reg" type="checkbox" />
            <span>Show regression line</span>
          </label>
        </div>
      </div>

      <div style="flex:1 1 520px; min-width:320px;">
        <div id="${uid}-svg-wrap" style="background:#fff; border:1px solid #eee; border-radius:8px; padding:12px;">
          <svg id="${uid}-svg" width="100%" height="420" viewBox="0 0 900 420" preserveAspectRatio="xMidYMid meet" style="display:block;"></svg>
        </div>
      </div>
    </div>

    <div id="${uid}-tooltip" role="status" aria-live="polite" style="position:absolute; display:none; pointer-events:none; background:rgba(0,0,0,0.8); color:white; padding:8px 10px; border-radius:6px; font-size:13px;"></div>
  `;

  host.appendChild(wrapper);

  // DOM references
  const svg = document.getElementById(`${uid}-svg`);
  const nInput = document.getElementById(`${uid}-n`);
  const nLabel = document.getElementById(`${uid}-n-label`);
  const yminInput = document.getElementById(`${uid}-ymin`);
  const yminLabel = document.getElementById(`${uid}-ymin-label`);
  const colorMode = document.getElementById(`${uid}-color`);
  const regToggle = document.getElementById(`${uid}-reg`);
  const tooltip = document.getElementById(`${uid}-tooltip`);

  // Basic state
  let state = {
    n: +nInput.value,
    ymin: +yminInput.value,
    colorMode: colorMode.value,
    showReg: regToggle.checked,
    data: []
  };

  // Utility: generate data
  function generateData(n) {
    const out = [];
    for (let i = 0; i < n; i++) {
      const x = Math.random() * 10; // 0..10
      // create a curved relationship with categories and noise
      const base = Math.sin(x) * 2 + x * 0.3;
      const noise = (Math.random() - 0.5) * 2;
      const y = base + noise;
      // bucket category by x
      const cat = Math.floor(x / (10 / 4)); // 0..3
      out.push({id: i, x, y, cat});
    }
    return out;
  }

  // Scales (simple linear mapping)
  function linearScale(domain0, domain1, range0, range1) {
    const d0 = domain0;
    const d1 = domain1;
    const r0 = range0;
    const r1 = range1;
    const m = (r1 - r0) / (d1 - d0 || 1);
    return v => r0 + (v - d0) * m;
  }

  // Color helpers
  const categoryColors = ['#1f77b4', '#ff7f0e', '#2ca02c', '#d62728'];
  function colorFor(d) {
    if (state.colorMode === 'single') return '#4e79a7';
    if (state.colorMode === 'byCategory') return categoryColors[d.cat % categoryColors.length];
    if (state.colorMode === 'byX') {
      // simple gradient blue->purple
      const t = d.x / 10;
      const r = Math.round(70 + t * 150);
      const g = Math.round(90 + t * 40);
      const b = Math.round(150 + t * 80);
      return `rgb(${r},${g},${b})`;
    }
    return '#333';
  }

  // Regression line (least-squares)
  function linearRegression(points) {
    const n = points.length;
    if (n === 0) return null;
    let sumX = 0, sumY = 0, sumXY = 0, sumXX = 0;
    for (const p of points) {
      sumX += p.x; sumY += p.y; sumXY += p.x * p.y; sumXX += p.x * p.x;
    }
    const slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX || 1);
    const intercept = (sumY - slope * sumX) / n;
    return {slope, intercept};
  }

  // Draw function (entire plot)
  function draw() {
    // Clear svg
    while (svg.firstChild) svg.removeChild(svg.firstChild);

    // Determine drawing size from viewBox
    const VBOX_W = 900, VBOX_H = 420;
    const margin = {top: 20, right: 18, bottom: 46, left: 56};
    const width = VBOX_W - margin.left - margin.right;
    const height = VBOX_H - margin.top - margin.bottom;

    const g = document.createElementNS('http://www.w3.org/2000/svg', 'g');
    g.setAttribute('transform', `translate(${margin.left},${margin.top})`);
    svg.appendChild(g);

    // Data filtering
    const filtered = state.data.filter(d => d.y >= state.ymin);

    // Domains
    const xMin = 0, xMax = 10;
    const yAll = state.data.map(d => d.y);
    const yMinGlobal = Math.min(...yAll, -5);
    const yMaxGlobal = Math.max(...yAll, 10);
    // Give a little padding
    const yPad = (yMaxGlobal - yMinGlobal) * 0.06;
    const yMin = Math.min(yMinGlobal, state.ymin) - yPad;
    const yMax = yMaxGlobal + yPad;

    const xScale = linearScale(xMin, xMax, 0, width);
    const yScale = v => height - linearScale(yMin, yMax, 0, height)(v);

    // Axes: simple ticks
    function drawAxes() {
      const axis = document.createElementNS('http://www.w3.org/2000/svg', 'g');
      axis.setAttribute('class', 'axes');

      // x axis line
      const xAxis = document.createElementNS('http://www.w3.org/2000/svg', 'line');
      xAxis.setAttribute('x1', 0); xAxis.setAttribute('x2', width);
      xAxis.setAttribute('y1', height); xAxis.setAttribute('y2', height);
      xAxis.setAttribute('stroke', '#ddd');
      axis.appendChild(xAxis);

      // y axis line
      const yAxis = document.createElementNS('http://www.w3.org/2000/svg', 'line');
      yAxis.setAttribute('x1', 0); yAxis.setAttribute('x2', 0);
      yAxis.setAttribute('y1', 0); yAxis.setAttribute('y2', height);
      yAxis.setAttribute('stroke', '#ddd');
      axis.appendChild(yAxis);

      // x ticks
      for (let i = 0; i <= 10; i += 2) {
        const tx = xScale(i);
        const t = document.createElementNS('http://www.w3.org/2000/svg', 'line');
        t.setAttribute('x1', tx); t.setAttribute('x2', tx);
        t.setAttribute('y1', height); t.setAttribute('y2', height + 6);
        t.setAttribute('stroke', '#ccc');
        axis.appendChild(t);

        const lab = document.createElementNS('http://www.w3.org/2000/svg', 'text');
        lab.setAttribute('x', tx); lab.setAttribute('y', height + 22);
        lab.setAttribute('text-anchor', 'middle');
        lab.setAttribute('font-size', '12');
        lab.setAttribute('fill', '#444');
        lab.textContent = String(i);
        axis.appendChild(lab);
      }

      // y ticks
      const yTicks = 5;
      for (let i = 0; i <= yTicks; i++) {
        const vy = yMin + (i / yTicks) * (yMax - yMin);
        const ty = yScale(vy);
        const t = document.createElementNS('http://www.w3.org/2000/svg', 'line');
        t.setAttribute('x1', -6); t.setAttribute('x2', 0);
        t.setAttribute('y1', ty); t.setAttribute('y2', ty);
        t.setAttribute('stroke', '#ccc');
        axis.appendChild(t);

        const lab = document.createElementNS('http://www.w3.org/2000/svg', 'text');
        lab.setAttribute('x', -10); lab.setAttribute('y', ty + 4);
        lab.setAttribute('text-anchor', 'end');
        lab.setAttribute('font-size', '12');
        lab.setAttribute('fill', '#444');
        lab.textContent = vy.toFixed(1);
        axis.appendChild(lab);
      }

      g.appendChild(axis);
    }

    drawAxes();

    // Points group
    const pointsG = document.createElementNS('http://www.w3.org/2000/svg', 'g');
    pointsG.setAttribute('class', 'points');
    g.appendChild(pointsG);

    // Draw points
    filtered.forEach(d => {
      const cx = xScale(d.x);
      const cy = yScale(d.y);
      const c = colorFor(d);

      const circle = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
      circle.setAttribute('cx', cx);
      circle.setAttribute('cy', cy);
      circle.setAttribute('r', 4.5);
      circle.setAttribute('fill', c);
      circle.setAttribute('stroke', '#fff');
      circle.setAttribute('stroke-width', '0.8');
      circle.style.cursor = 'pointer';

      // attach data for event handlers
      circle.__data = d;

      // Hover interactions
      circle.addEventListener('mouseenter', (ev) => {
        const data = ev.target.__data;
        tooltip.style.display = 'block';
        tooltip.innerHTML = `<strong>x:</strong> ${data.x.toFixed(2)} <br/><strong>y:</strong> ${data.y.toFixed(2)} <br/><strong>cat:</strong> ${data.cat}`;
      });
      circle.addEventListener('mousemove', (ev) => {
        tooltip.style.left = (ev.pageX + 12) + 'px';
        tooltip.style.top = (ev.pageY - 28) + 'px';
      });
      circle.addEventListener('mouseleave', () => {
        tooltip.style.display = 'none';
      });

      pointsG.appendChild(circle);
    });

    // Regression line if requested and there are points
    if (state.showReg && filtered.length >= 2) {
      const {slope, intercept} = linearRegression(filtered);
      // x at domain start and end
      const x0 = xMin;
      const x1 = xMax;
      const y0 = slope * x0 + intercept;
      const y1 = slope * x1 + intercept;
      const line = document.createElementNS('http://www.w3.org/2000/svg', 'line');
      line.setAttribute('x1', xScale(x0)); line.setAttribute('y1', yScale(y0));
      line.setAttribute('x2', xScale(x1)); line.setAttribute('y2', yScale(y1));
      line.setAttribute('stroke', '#b550e6');
      line.setAttribute('stroke-width', '2');
      line.setAttribute('stroke-linecap', 'round');
      line.setAttribute('opacity', '0.9');
      g.appendChild(line);
    }

    // Draw a small legend for categories when in categorical mode
    if (state.colorMode === 'byCategory') {
      const legend = document.createElementNS('http://www.w3.org/2000/svg', 'g');
      legend.setAttribute('transform', `translate(${width - 60}, 6)`);
      for (let i = 0; i < categoryColors.length; i++) {
        const y = i * 18;
        const rect = document.createElementNS('http://www.w3.org/2000/svg', 'rect');
        rect.setAttribute('x', 0); rect.setAttribute('y', y); rect.setAttribute('width', 12); rect.setAttribute('height', 12);
        rect.setAttribute('fill', categoryColors[i]);
        legend.appendChild(rect);

        const lab = document.createElementNS('http://www.w3.org/2000/svg', 'text');
        lab.setAttribute('x', 18); lab.setAttribute('y', y + 10);
        lab.setAttribute('font-size', '12'); lab.setAttribute('fill', '#333');
        lab.textContent = `cat ${i}`;
        legend.appendChild(lab);
      }
      g.appendChild(legend);
    }
  }

  // Debounce helper
  function debounce(fn, wait = 120) {
    let t;
    return (...args) => {
      clearTimeout(t);
      t = setTimeout(() => fn(...args), wait);
    };
  }

  // Update state and redraw
  const redraw = debounce(() => {
    // update labels
    nLabel.textContent = String(state.n);
    yminLabel.textContent = String(state.ymin.toFixed(1));
    // regenerate data if n changed
    if (state.data.length !== state.n) {
      state.data = generateData(state.n);
    }
    draw();
  }, 50);

  // Event listeners
  nInput.addEventListener('input', (e) => {
    state.n = +e.target.value;
    nLabel.textContent = String(state.n);
    // regenerate and draw (debounced)
    state.data = generateData(state.n);
    redraw();
  });

  yminInput.addEventListener('input', (e) => {
    state.ymin = +e.target.value;
    yminLabel.textContent = String(state.ymin.toFixed(1));
    redraw();
  });

  colorMode.addEventListener('change', (e) => {
    state.colorMode = e.target.value;
    redraw();
  });

  regToggle.addEventListener('change', (e) => {
    state.showReg = e.target.checked;
    redraw();
  });

  // Responsive: redraw when the svg element resizes (simple approach)
  const resizeObserver = new ResizeObserver(() => {
    // Keep viewBox constant, but redrawing keeps pixel-perfect axis ticks and positions
    redraw();
  });
  resizeObserver.observe(svg);

  // Initial data and render
  state.data = generateData(state.n);
  redraw();
})();
